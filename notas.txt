CREATE TABLE cobros (
  id_cobro INT AUTO_INCREMENT PRIMARY KEY,
  id_reparacion INT NOT NULL,
  fecha_cobro DATE NOT NULL DEFAULT CURRENT_DATE,
  monto_cobrado DECIMAL(10,2) NOT NULL,
  medio_pago VARCHAR(50) DEFAULT NULL,
  nro_comprobante
  observaciones TEXT DEFAULT NULL,
  FOREIGN KEY (id_reparacion) REFERENCES reparaciones(id_reparacion)
);


id_cobro: identificador único.

id_reparacion: lo que vincula este cobro con una reparación.

fecha_cobro: se completa automáticamente si no lo especificás.

monto_cobrado: cuánto pagó el cliente.

medio_pago: por si querés registrar si fue efectivo, transferencia, etc.

observaciones: campo opcional para anotar cualquier detalle (comprobante, aclaración, etc.).


ALTER TABLE atencion_presupuesto DROP COLUMN monto_pago_inicial;

SELECT
  ap.presupuesto,
  ap.monto_pago_inicial + IFNULL(SUM(c.monto_cobrado), 0) AS total_pagado,
  ap.presupuesto - (ap.monto_pago_inicial + IFNULL(SUM(c.monto_cobrado), 0)) AS saldo_restante
FROM atencion_presupuesto ap
LEFT JOIN cobros c ON c.id_reparacion = ap.id_reparacion
WHERE ap.id_reparacion = ?
GROUP BY ap.id_reparacion;


 Consulta simple para ver lo pagado por reparación


SELECT 
  r.id_reparacion,
  ap.presupuesto,
  IFNULL(SUM(c.monto_cobrado), 0) AS total_pagado,
  (ap.presupuesto - IFNULL(SUM(c.monto_cobrado), 0)) AS saldo_pendiente
FROM reparaciones r
JOIN atencion_presupuesto ap ON ap.id_reparacion = r.id_reparacion
LEFT JOIN cobros c ON c.id_reparacion = r.id_reparacion
GROUP BY r.id_reparacion;

 ¿Qué ventajas tiene esta estructura?
Podés manejar fácilmente reparaciones con múltiples pagos (pagos parciales).

Separás bien lo técnico (reparación) de lo administrativo (presupuesto y cobros).

Podés extender fácilmente: agregar facturación, comprobantes, etc.

Podés filtrar reparaciones con presupuesto aprobado pero sin pagos, o con saldo pendiente.
